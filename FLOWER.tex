\Procedure{FlowerController}{$sensorReading$}
\If {$hitWaypoint$} \Return getHome()
\Else $ $ \Return findLight(sensorReading)

\EndProcedure


\Procedure{findLight}{$sensorReading$}
\If {$\text{-}\text{-}movesFowardRemaining > \textit{0}$} 
\If {$sensorReading > lightHitValue $} 
\State $hitWaypoint \gets \textit{true}$
\State $resetState()$
\EndIf
\State \Return fowards()
\ElsIf {$\text{-}\text{-}looksRemaining >  \textit{0}$}  \Return look()
\ElsIf {$\text{-}\text{-}orientsRemaining >  \textit{0}$} \Return orient()
\Else 
\State $resetState()$
\State \Return fowards()
\EndIf
\EndProcedure

\Procedure{getHome}{$sensorReading$}
\If {$ringRemaining == empty$} 
\State $ringRemaining \gets \uppi \textit{ * RingDiameter}$
\EndIf
\If {$\text{-}\text{-}reversesRemaining > \textit{0}$} \Return eject()
\ElsIf {$\text{-}\text{-}ringRemaining > 0$}
\Return moveInRing()
\Else 
\State $\textit{ hitWaypoint} \gets \textit{true}$
\State \Return $ findLight(sensorReading)$
\EndIf
\EndProcedure

\end{algorithmic}
